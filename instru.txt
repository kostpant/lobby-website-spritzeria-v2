The scroll event listener is completely missing 
from the file. There is no addEventListener scroll 
anywhere. That is why the animation does not play.

Find the JavaScript section in the file.
Find where the frames are loaded and drawFrame exists.
Add this scroll listener right after the 
frame preloading code. Do not change anything else:

window.addEventListener('scroll', function() {
  var wrapper = document.getElementById('hero-wrapper');
  if (!wrapper) return;
  var rect = wrapper.getBoundingClientRect();
  var scrolled = -rect.top;
  var total = wrapper.offsetHeight - window.innerHeight;
  if (scrolled < 0) scrolled = 0;
  if (total <= 0) return;
  var progress = scrolled / total;
  if (progress > 1) progress = 1;
  var idx = Math.floor(progress * (TOTAL_FRAMES - 1));
  if (idx > TOTAL_FRAMES - 1) idx = TOTAL_FRAMES - 1;
  draw(idx);
}, { passive: true });

Add only this. Nothing else changes.
